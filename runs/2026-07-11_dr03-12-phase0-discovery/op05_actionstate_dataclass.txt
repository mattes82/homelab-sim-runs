def compute_eligibility(row, proposed_mode: str) -> tuple[bool, list[str]]:
    """Hard structural gate (Rules of Engagement 1.4). Not overridable."""
    reasons: list[str] = []
    if not row["reversible"]:
        reasons.append("not fully reversible")
    if not row["undo_supported"]:
        reasons.append("no undo support")
    if row["touches_money"]:
        reasons.append("touches money")
    if row["external_comm"]:
        reasons.append("external communication")
    if row["performs_delete"]:
        reasons.append("performs delete")
    if row["ceiling_risk_class"] == "HARD_STOP":
        reasons.append("ceiling risk class HARD_STOP")
    if proposed_mode not in AUTONOMY_MODES:
        reasons.append(f"target mode {proposed_mode} is not an autonomy mode")
    return (len(reasons) == 0, reasons)
