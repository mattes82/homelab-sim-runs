// known-sender Pre-Filter — Klassifikator v1.5
// Laeuft NACH dem HTTP-Node 'known-senders laden'; $json ist daher die
// known-senders-Map, nicht die Mail. Mail-Daten daher explizit aus
// 'Mail klassifizieren' referenzieren. Feldname der Absenderdomain: from_domain.
const FROM_DOMAIN_FIELD = "from_domain";

const knownSenders = $('known-senders laden').first().json || {};
const mail = $('Mail klassifizieren').first().json;
const fromDomain = String(mail[FROM_DOMAIN_FIELD] || "").toLowerCase().trim();

const matchedCategory = knownSenders[fromDomain] || null;
const isMatch = matchedCategory !== null;

return [{
  json: {
    ...mail,
    _known_sender_match: isMatch,
    _known_sender_category: matchedCategory,
    // Bei known-sender-Treffer den Paperless-Branch umgehen, damit der Override
    // den Dispatch-Switch erreicht (Pre-Filter hat Vorrang vor der Heuristik).
    paperless_candidate: isMatch ? false : mail.paperless_candidate,
  }
}];
