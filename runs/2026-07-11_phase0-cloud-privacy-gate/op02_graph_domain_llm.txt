## 'Domain-Check' jsCode:
const domain = $('Mail klassifizieren').first().json.from_domain;
const known = $('Shop-Liste laden').first().json.known_shops;
const domain_known = Array.isArray(known) && known.includes(domain);
return [{ json: { domain_known } }];
