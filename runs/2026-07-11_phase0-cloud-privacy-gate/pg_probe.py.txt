def db():
    return sqlite3.connect(f"file:{DB}?mode=ro", uri=True, timeout=15)

def max_exec_id(wf_id):
    con = db(); r = con.execute("SELECT MAX(id) FROM execution_entity WHERE workflowId=?", (wf_id,)).fetchone(); con.close()
    return r[0] or 0

def wait_new_exec(wf_id, prev, timeout=45):
    t0 = time.time()
    while time.time() - t0 < timeout:
        con = db()
        rows = con.execute("SELECT id, finished, status FROM execution_entity WHERE workflowId=? AND id>? ORDER BY id", (wf_id, prev)).fetchall()
        con.close()
        for (eid, fin, st) in rows:
            if fin == 1 or st in TERMINAL:
                return eid, st
        time.sleep(1)
    return None, None

def unflatten(data_str):
    arr = json.loads(data_str); resolved = [None] * len(arr)
    def resolve(i):
        if resolved[i] is not None: return resolved[i]
        val = arr[i]
        if isinstance(val, dict):
            out = {}; resolved[i] = out
            for k, v in val.items(): out[k] = deref(v)
            return out
        if isinstance(val, list):
            out = []; resolved[i] = out
            for v in val: out.append(deref(v))
            return out
        resolved[i] = val; return val
    def deref(v):
        if isinstance(v, str) and v.isdigit() and int(v) < len(arr): return resolve(int(v))
        return v
    return resolve(0)

def exec_data(eid):
    con = db(); r = con.execute("SELECT data FROM execution_data WHERE executionId=?", (eid,)).fetchone(); con.close()
    return unflatten(r[0]) if r else None

def node_out(edata, node_name):
    try:
        run = edata["resultData"]["runData"][node_name][0]
        return run["data"]["main"][0][0]["json"]
    except (KeyError, IndexError, TypeError):
        return None

def post(url, body):
    req = urllib.request.Request(url, data=json.dumps(body).encode(), method="POST",
                                 headers={"Content-Type": "application/json"})
    try:
        with urllib.request.urlopen(req, timeout=20) as r:
            return r.status
    except urllib.error.HTTPError as e:
        return e.code
