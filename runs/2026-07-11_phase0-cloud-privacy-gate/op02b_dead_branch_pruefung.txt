## 'Normalisierung Kategorie' jsCode (dispatch-m-v1.5):
// Normalisierung Kategorie — Klassifikator v1.5
// Erweitert die FIX-M Normalisierung (vor 'Dispatch: Kategorie') um:
//  - known-sender Pre-Filter-Vorrang (Override der Heuristik)
//  - classification_source / low_confidence / raw_category
//  - Markierung nicht zuordenbarer Heuristik-Kategorie ('Allgemein') fuer ntfy-Alert
//
// WHITELIST = die im Dispatch-Switch definierten 7 Kategorien mit eigenem Pfad.
// Alles andere -> Review / Sonstiges (kein eigener Action-Pfad, SHADOW-Review).
const WHITELIST = ["Lieferung / Versand", "Newsletter / Werbung", "Rechnung / Finanzen", "Account / Sicherheit", "Vertrag", "Termin / Kalender", "Retoure"];
const REVIEW = 'Review / Sonstiges';

for (const item of $input.all()) {
  const j = item.json;
  const rawCategory = j.category;          // Output der Heuristik 'Mail klassifizieren'
  j.raw_category = rawCategory;

  if (j._known_sender_match && j._known_sender_category) {
    // known-sender Pre-Filter hat Vorrang vor der Heuristik.
    j.category = j._known_sender_category;
    j.classification_source = 'known-sender';
    j.low_confidence = false;
  } else {
    j.classification_source = 'classifier';
    // 'Allgemein' = Heuristik konnte nicht zuordnen -> low confidence + Alert.
    j.low_confidence = (rawCategory === 'Allgemein');
  }

  // Whitelist-Routing: nicht-geroutete Kategorien -> Review / Sonstiges.
  if (!WHITELIST.includes(j.category)) {
    j.category_original = rawCategory;
    j.category = REVIEW;
  }

  // interne Pre-Filter-Flags entfernen.
  delete j._known_sender_match;
  delete j._known_sender_category;
}
return $input.all();
